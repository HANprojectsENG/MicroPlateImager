132  if N > 1 and (N & 1) == 1:

134  mode='constant') # Assuming N is odd

141 def find1DGrid(data, N):
142  if N <= 1:
143  raise ValueError('findGrid parameter <= 1')
144  if (N & 1) != 1: # enforce N to be odd
145  N += 1
146  gridSmoothKsize = N
147  gridMinSegmentLength = 10*N
148 
149  # High-pass filter, to suppress uneven illumination
150  data = np.abs(data - moving_average(data, int(3*N)))
151  data[:N] = 0 # cut off MA artifacts
152  data[-N:] = 0 # cut off MA artifacts, why not -(N-1)/2?? ??
153  smooth_data = moving_average(data, gridSmoothKsize)
154  smooth_data = smooth_data - np.mean(smooth_data)
155  mask_data = np.zeros(data.shape, dtype='bool') # mask grid lines
156  mask_data[np.where(smooth_data < 0)[0]] = True
157 
158  # Now filter mask_data based on segment length and suppress too short segments
159  prev_x = False
160  segmentLength = 0
161  segmentList = []
162  for index, x in enumerate(mask_data):
163  if x: # segment
164  segmentLength += 1
165  elif x != prev_x: # falling edge
166  if segmentLength < gridMinSegmentLength: # suppress short segments
167  mask_data[index - segmentLength: index] = False
168  # print(diff(data[index - segmentLength:index]))
169  else:
170  segmentList.append((index - segmentLength, segmentLength)) # Save segment start and length
171  segmentLength = 0 # reset counter
172  prev_x = x
173
