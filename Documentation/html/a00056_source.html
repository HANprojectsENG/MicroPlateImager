183 def raw_resolution(resolution, splitter=False):
184  """
185  Round a (width, height) tuple up to the nearest multiple of 32 horizontally
186  and 16 vertically (as this is what the Pi's camera module does for
187  unencoded output).
188  """
189  width, height = resolution
190  if splitter:
191  fwidth = (width + 15) & ~15
192  else:
193  fwidth = (width + 31) & ~31

195  return fwidth, fheight
